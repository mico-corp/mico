134 unsigned gpioHardwareRevision(void)
135 {
136  static unsigned rev = 0;
137 
138  FILE * filp;
139  char buf[512];
140  char term;
141  int chars=4; /* number of chars in revision string */
142 
143  if (rev) return rev;
144 
145  piModel = 0;
146 
147  filp = fopen ("/proc/cpuinfo", "r");
148 
149  if (filp != NULL)
150  {
151  while (fgets(buf, sizeof(buf), filp) != NULL)
152  {
153  if (piModel == 0)
154  {
155  if (!strncasecmp("model name", buf, 10))
156  {
157  if (strstr (buf, "ARMv6") != NULL)
158  {
159  piModel = 1;
160  chars = 4;
161  }
162  else if (strstr (buf, "ARMv7") != NULL)
163  {
164  piModel = 2;
165  chars = 6;
166  }
167  else if (strstr (buf, "ARMv8") != NULL)
168  {
169  piModel = 2;
170  chars = 6;
171  }
172  }
173  }
174 
175  if (!strncasecmp("revision", buf, 8))
176  {
177  if (sscanf(buf+strlen(buf)-(chars+1),
178  "%x%c", &rev, &term) == 2)
179  {
180  if (term != '\n') rev = 0;
181  }
182  }
183  }
184 
185  fclose(filp);
186  }
187  return rev;
188 }
