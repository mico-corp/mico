49  struct ConfigParameterDef {
50  enum eParameterType { BOOLEAN, INTEGER, DECIMAL, STRING, PATH, OPTIONS };
51  std::string name_;
52  eParameterType type_;
53  boost::any value_;
54  bool asBool() const { assert(type_ == eParameterType::BOOLEAN); return boost::any_cast<bool>(value_); };
55  int asInteger() const { assert(type_ == eParameterType::INTEGER); return boost::any_cast<int>(value_); };
56  float asDecimal() const { assert(type_ == eParameterType::DECIMAL); return boost::any_cast<float>(value_); };
57  std::string asString() const { assert(type_ == eParameterType::STRING); return boost::any_cast<std::string>(value_); };
58  fs::path asPath() const { assert(type_ == eParameterType::PATH); return boost::any_cast<fs::path>(value_); };
59  std::vector<std::string> asOptions() const { assert(type_ == eParameterType::OPTIONS); return boost::any_cast<std::vector<std::string>>(value_); };
60  };
